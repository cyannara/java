```java
public class NumberUtil {
	public void sum() {
		int a = 10;
		int b = 20;
		int result = 0;
		result = a + b;
		System.out.println(result);
	}

	public void sum(int a, int b) {
		int result = 0;
		result = a + b;
		System.out.println(result);
	}

	public int sumRe(int a, int b) {
		int result = 0;
		result = a + b;
		return result;
	}

	public void sumRef(Integer a, Integer b, Integer result) {
		result = a + b;
	}
}

public class Main(){
	public void static main(String arg[]){
		Integer a = 10;
		Integer b = 10;
		Integer result = 0;
		NumberUtil nu = new NumberUtil();

		nu.sumRef(a, b, result);
		System.out.println(result);
	}
}
```

```java
// 인수와 리턴값이 없는 경우
	public void helloUpper() {
		String str = "hello";
		String result = str.toUpperCase();
		System.out.println(result);
	}

	// 인수가 있는 경우
	public void upper(String str) {
		str.toUpperCase();
		System.out.println(str);
	}

	// 인수와 리턴값이 있는 경우
	public String upperReturn(String str) {
		String result = str.toUpperCase();
		return result;
	}

	//참조변수
	public void upperArrayRef(String[] strs) {
		for(int i=0; i<strs.length; i++) {
			strs[i] = strs[i].toUpperCase();
		}
	}

	public void printArray(String[] strs) {
		for(String str : strs) {
			System.out.println(str);
		}
	}
```
